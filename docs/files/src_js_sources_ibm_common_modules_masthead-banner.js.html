                        /**
                         * Masthead Banner, activated by the `masthead-banner.json` data file
                         *
                         * @class IBMCore.common.module.mastheadbanner
                         */
                        (function($, IBM) {
                          var me = IBM.namespace(IBM, 'common.module.mastheadbanner');
                        
                          /**
                           * Sets the data url for the banner
                           * @type {string}
                           * @private
                           */
                          var _dataUrl =
                            'https://www.ibm.com/common/v18/mastheadbanner/masthead-banner.json';
                        
                          /**
                           * Key name for caching the banner content in session storage
                           * @type {string}
                           * @private
                           */
                          var _sessionStorageKey = 'v18mastheadbanner';
                        
                          /**
                           Automated callback from subscription to meta 'dataReady' event publisher.
                           @method autoInit
                           **/
                          me.autoInit = function() {
                            _initBanner();
                          };
                        
                          /**
                           * Fetches the banner data before injecting
                           * @private
                           */
                          function _initBanner() {
                            var bannerData = JSON.parse(sessionStorage.getItem(_sessionStorageKey));
                        
                            if (bannerData) {
                              _injectBanner(bannerData);
                            } else {
                              $.getJSON(_dataUrl, function(response) {
                                sessionStorage.setItem(_sessionStorageKey, JSON.stringify(response));
                                _injectBanner(response);
                              });
                            }
                          }
                        
                          /**
                           * Checks if the wildcard list appears in the pathname
                           * @param {Array} list array of blacklist path substrings
                           * @returns {boolean}
                           * @private
                           */
                          function _checkWildcard(list) {
                            var hasWildcard = false;
                            list.forEach(function(value) {
                              if (location.pathname.indexOf(value) !== -1) {
                                hasWildcard = true;
                              }
                            });
                            return hasWildcard;
                          }
                        
                          /**
                           * Checks if the criteria is met before injecting the banner:
                           * - bannerActive is true
                           * - current location.pathname is in the whitelist
                           * - lang content (lc-cc) exists
                           *
                           * @param data
                           * @private
                           */
                          function _injectBanner(data) {
                            var lc = IBM.common.util.meta.getPageLc();
                            var cc = IBM.common.util.meta.getPageCc();
                            var lang = lc + '-' + cc;
                        
                            if (
                              data.thinkBanner &&
                              data.thinkBanner.active &&
                              data.thinkBanner.blacklist.indexOf(location.pathname) === -1 &&
                              !_checkWildcard(data.thinkBanner.blacklistWildcard || []) &&
                              data.thinkBanner.blacklistCountry.indexOf(cc) === -1 &&
                              data.thinkBanner.content[lc] &&
                              $('#think-banner-container').length === 0
                            ) {
                              _renderThinkBanner(
                                data.thinkBanner.content[lc],
                                data.thinkBanner.countdownDate
                              );
                            } else if (
                              data.bannerActive &&
                              data.whitelist.indexOf(location.pathname) !== -1 &&
                              data.content[lang] &&
                              $('#masthead-banner').length === 0
                            ) {
                              _renderBanner(data.content[lang]);
                            }
                          }
                        
                          /**
                           * Renders the banner above the masthead
                           * @param {object} data Banner data
                           * @private
                           */
                          function _renderBanner(data) {
                            $('body').prepend(
                              '<a id="masthead-banner" class="ibm-blocklink" href="' +
                                data.ctaURL +
                                '">' +
                                '<div id="masthead-banner-content">' +
                                '<div id="masthead-banner-desktop">' +
                                '<div class="masthead-banner--flex masthead-banner--align-center">' +
                                '<img class="masthead-banner__icon" src="' +
                                data.iconUrl +
                                '" />' +
                                '<span class="ibm-bold ibm-h4-small">' +
                                data.bannerText +
                                '</span>' +
                                '</div>' +
                                '<div class="cta">' +
                                '<span class="cta__text">' +
                                data.ctaLabel +
                                '</span>' +
                                '</div>' +
                                '</div>' +
                                '<!-- Mobile banner -->' +
                                '<div id="masthead-banner-mobile">' +
                                '<div class="masthead-banner--flex masthead-banner--align-center">' +
                                '<img class="masthead-banner__icon" src="' +
                                data.iconUrl +
                                '" />' +
                                '<span class="ibm-bold" style="word-wrap: break-word;">' +
                                data.bannerTextMobile +
                                '</span>' +
                                '<span class="forward-icon"></span>' +
                                '</div>' +
                                '</div>' +
                                '<!-- Mobile banner -->' +
                                '</div>' +
                                '</a>'
                            );
                          }
                        
                          /**
                           * Renders the Think Banner
                           * @param {object} data Banner data
                           * @param {string} countdown Date to countdown to
                           * @private
                           */
                          function _renderThinkBanner(data, countdown) {
                            var countDownDate = new Date(countdown).getTime();
                            var now = new Date().getTime();
                            var distance = countDownDate - now;
                            var days = Math.ceil(distance / (1000 * 60 * 60 * 24));
                            if (days <= 0) {
                              days = '';
                            }
                            var line1 = days === '' ? data.bannerTextLive1 : data.bannerTextLine1;
                            var line2 = days === '' ? data.bannerTextLive2 : data.bannerTextLine2;
                            if (line2 !== '') {
                              line2 = '<br/>' + line2;
                            }
                            var ctaLabel = days === '' ? data.ctaLabelLive : data.ctaLabel;
                        
                            $('body').prepend(
                              '<div id="think-banner-container">' +
                                '<a id="think-banner-link" class="ibm-blocklink ibm-alternate-background" href="' +
                                data.ctaUrl +
                                '">' +
                                '<div id="think-banner-content">' +
                                '<div class="ibm-columns ibm-padding-bottom-0">' +
                                '<div id="think-banner-desktop">' +
                                '<div class="video-background">' +
                                '<video autoplay muted>' +
                                '<source type="video/mp4" src="' +
                                data.videoUrl +
                                '" />' +
                                '</video>' +
                                '</div>' +
                                '<div class="flex align-center think-counter__container">' +
                                '<div class="think-counter">' +
                                days +
                                '</div>' +
                                '<div>' +
                                '<span class="ibm-bold ibm-h4-small counter-text">' +
                                line1 +
                                line2 +
                                '</span>' +
                                '</div>' +
                                '</div>' +
                                '<div class="cta">' +
                                '<span class="cta__text">' +
                                ctaLabel +
                                '</span>' +
                                '</div>' +
                                '</div>' +
                                '<div id="think-banner-mobile">' +
                                '<div class="flex align-center space-between">' +
                                '<div class="flex align-center space-between">' +
                                '<div class="flex align-center">' +
                                '<div class="think-counter">' +
                                days +
                                '</div>' +
                                '<div>' +
                                '<span class="ibm-bold ibm-h4-small counter-text">' +
                                line1 +
                                line2 +
                                '</span>' +
                                '</div>' +
                                '</div>' +
                                '<span class="forward-icon"></span>' +
                                '</div>' +
                                '</div>' +
                                '</div>' +
                                '</div>' +
                                '</div>' +
                                '</a>' +
                                '</div>'
                            );
                          }
                        
                          IBM.common.meta
                            .subscribe('dataReady', 'mastheadbanner', me.autoInit)
                            .runAsap(me.autoInit);
                        })(jQuery, IBMCore);
