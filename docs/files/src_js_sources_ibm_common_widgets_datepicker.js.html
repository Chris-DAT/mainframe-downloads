File: src/js/sources/ibm/common/widgets/datepicker.js

                        /**
                        
                        	Date picker plug-in pre-processor. 
                        	<br />
                        	<br />Sets some defaults and other random processing, then it inits the widget.
                        	<br />v18 uses the pickadate/time plug-in: <a href="http://amsul.ca/pickadate.js/" target="_blank">http://amsul.ca/pickadate.js/</a>
                        	<br />For all options, detailed documentation, APIs, and advanced usage read the  <a href="http://amsul.ca/pickadate.js/" target="_blank">http://amsul.ca/pickadate.js/</a> documentation.
                        
                        	<br />
                        	<br />Usage
                        	   
                            <input type="text" value="" data-widget="datepicker">
                        
                        	<br />__Event usage__: Subscribe to the IBM widget init event that fires on your DOM element to know when the widget has been initialized and you can access its APIs.
                        
                        	<br>
                        	<br>
                        	__Special case:__ In scenarios where you dynamically inject HTML post page load, you can initialize the widget on your injected container like this:
                        	<br>- Passing in a string with the ID, or the DOM element pointer.
                        	<br>- Optionally pass any settings you want (instead of using data-attributes on the HTML).
                        		
                        	IBMCore.common.widget.datepicker.init("yourFieldId", optionalSettingsObject);
                        	 // Or
                        	IBMCore.common.widget.datepicker.init(yourFieldDOMelement, optionalSettingsObject);
                        
                        	__To destroy the widget__ and return back the original HTML that the widget initialized on, call the common widget destroy method on your element:
                        		
                        	jQuery("<your_input_selector>").data("widget").destroy();
                        
                        	@class IBMCore.common.widget.datepicker
                        
                        **/
                        
                        
                        (function ($, IBM) {
                        
                        	var me = IBM.namespace(IBM, "common.widget.datepicker");
                        
                        	/**
                        		Init function called on load for any date or time pickers that exist on the page at time of load.
                        		<br />Inits the pickatime/date plug-in using any user settings to override default that v18 sets.
                        	
                        		@method init
                        		@param el {DOM element} The &lt;input> element you want to add the widget to.
                        		@param [settings] {Object} Settings to override defaults and element's @data-xxxx attributes.
                        	**/
                        	me.init = function (el, settings) {
                        		var $elem = $(el),
                        			$origEl = $elem.clone(true),
                        			config = {},
                        			defaults = {
                        				closeOnClear: true,
                        				closeOnSelect: true,
                        				enabletranslations: true,
                        				firstDay: 0,
                        				format: "yyyy/mm/dd",
                        				includeicon: true,
                        				klass: {
                        					buttonClear: "ibm-btn-pri ibm-btn-blue-50 ibm-btn-small",
                        					buttonClose: "ibm-btn-pri ibm-btn-blue-50 ibm-btn-small ibm-fright",
                        					buttonToday: "ibm-btn-pri ibm-btn-blue-50 ibm-btn-small ibm-fleft",
                        					navPrev: "picker__nav--prev ibm-chevron-left-bold-link ibm-textcolor-gray-50",
                        					navNext: "picker__nav--next ibm-chevron-right-bold-link ibm-textcolor-gray-50",
                        					navDisabled: "picker__nav--disabled ibm-hasicon ibm-textcolor-gray-20"
                        				},
                        				localeToUse: IBM.common.util.meta.getPageCc() + IBM.common.util.meta.getPageLc(),
                        				onClose: function () {
                        					this.$holder.blur();
                        				},
                         				selectMonths: true,
                        				selectYears: true
                        			},
                        			pickerWidget;
                        
                        		// Merge in configuration to use, then init plug-in with them.
                        		config = $.extend({}, defaults, $elem.data() || {}, settings);
                        		
                        		// Use es/es for all ES languages.
                        		if (IBM.common.util.meta.getPageLc() === "es") {
                        			config.localeToUse = "eses";
                        		}
                        
                        
                        		// This is the actual function that inits the widget. We have to get the translations first (ajax promise below), 
                        		// then call this after translations are retrieved.
                        		function initWidget () {
                        			// Init the actual widget and add our standard class to it.
                        			pickerWidget = $elem.pickadate(config).pickadate("picker");
                        
                        			$elem.data("widget", pickerWidget);
                        
                        			// Add common destroy function API.
                        			// Destroys the widget and puts back the original HTML element that existed before the widget initialized.
                        			$elem.data("widget").destroy = function () {
                        				$elem.data("widget").stop();
                        				$elem.data("widget").$icon.remove();
                        				$elem.replaceWith($origEl);
                        				return $origEl;
                        			};
                        		
                        			// Find select lists inside the datepicker widget elment root and prevent/disable 
                        			//  the select2 widget on select lists inside the date picker.
                        			pickerWidget.$root.find("select").attr("data-init", false);
                        
                        			// If they didn't opt-out of showing the icon, add it/overlay it on top of the input field and on 
                        			//  click of the icon, trigger a click on the input so the datepicker opens.
                        			if (config.includeicon) {
                        				$elem.data("widget").$icon = $('<a class="ibm-calendar-link" href="#" tabindex="-1">&nbsp;</a>').insertAfter($elem).on("click", function (evt) {
                        					evt.preventDefault();
                        					evt.stopPropagation();
                        					pickerWidget.open();
                        				});
                        			}
                        
                        			// Add processed state (class) and fire common init event on the HTML element.
                        			IBM.common.widget.manager.dispatchInitEvent($elem[0]);
                        		}
                        		
                        		// Request the translations, since defaults are set as EN we don't really care if this fails.
                        		// If it does fail (silently) it just means the locale xlations didn't exist and will fallback to EN.
                        		if (config.enabletranslations) {
                        			$.ajax({
                        				url: IBM.common.util.config.get("datepickerDataUrl") + config.localeToUse + ".js",
                        				dataType: "script",
                        				cache: true
                        			}).always(function () {
                        				initWidget();
                        			});
                        		}
                        		else {
                        			initWidget();
                        		}
                        	};
                        
                        })(jQuery, IBMCore);
