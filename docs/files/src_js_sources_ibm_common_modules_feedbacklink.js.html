                        /**
                        
                        	Feedback link in footer
                        	<br />
                        	<br />This injects the feedback link in the footer and calls the feedback function onclick. That's it.
                        	<br />
                        
                        	@class Other IBM - footer feedback link
                          @deprecated
                        
                        **/
                        
                        (function($, IBM) {
                          var me = IBM.namespace(IBM, 'common.module.feedbacklink'),
                            defaultLinktext = 'Feedback',
                            footerSubscription = false,
                            settings = {
                              surveyId: 'web-exp',
                              metaData: '',
                            };
                        
                          /**
                        		Callback from subscription to meta 'dataReady'.
                        		<br />If feedbacklink (footer) is enabled (default all pages) inject it.
                        		<br />Only allowed to work on *.ibm.com pages per DOU with them.
                        
                        		@method autoInit
                        	**/
                          me.autoInit = function() {
                            if (window.location.hostname.indexOf('.ibm.com') === -1) {
                              return;
                            }
                            if (IBM.common.util.config.isEnabled('feedbacklink')) {
                              me.init();
                            }
                          };
                        
                          /**
                        		Subscribes to footer and sets flag if we didn't already subscribe,
                        		 and once footer is ready, injects the "feedback" link.
                        
                        		@method init
                        	**/
                          me.init = function() {
                            if (!footerSubscription) {
                              IBM.common.module.footer
                                .subscribe('ready', 'feedbacklink', injectFeedbackLink)
                                .runAsap(injectFeedbackLink);
                              footerSubscription = true;
                            }
                          };
                        
                          /**
                        		Inject the "feedback" link into the footer and bind it to init the survey (call the API) on click.
                        		<br>Also re-merges settings from page owner config in case they changed on a locale or other dynamic page change.
                        
                        		@method injectFeedbackLink
                        	**/
                          me.injectFeedbackLink = injectFeedbackLink;
                          function injectFeedbackLink() {
                            // Set the link text. We don't run until footer exists, and footer doesn't run until translations are ready.
                            // Therefore, tranlations are already ready at this point.
                            var linkText =
                                IBM.common.translations.data.v18main.misc.feedback || defaultLinktext,
                              $footerUl = $(
                                document.getElementsByClassName('ibm-footer-corporate-links')[0]
                              ).find('ul');
                        
                            // Remove the one that was there (for case of dynamic locale change via meta util) and inject new one.
                            $footerUl.children('.ibm-feedbacklink').remove();
                        
                            $('<li class="ibm-feedbacklink"><a href="#">' + linkText + '</a></li>')
                              .appendTo($footerUl)
                              .on('click', function(evt) {
                                evt.preventDefault();
                        
                                // RE-merge locally set configs for cases where meta data was set after initial page load,
                                //  since the data is only used onclick on the footer link.
                                $.extend(true, settings, IBM.common.util.config.get('feedbacklink'));
                        
                                IBM.common.widget.surveyccfintercept.autoInit();
                              });
                          }
                        
                          // Medallia is only set (contractual) to work on .ibm.com domain, so don't even bother
                          //  subscribing and auto-initing if it's not.
                          if (window.location.hostname.indexOf('.ibm.com') > 0) {
                            IBM.common.meta
                              .subscribe('dataReady', 'feedbacklink', me.autoInit)
                              .runAsap(me.autoInit);
                          }
                        })(jQuery, IBMCore);
